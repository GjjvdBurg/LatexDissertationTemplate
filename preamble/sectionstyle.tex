% section formatting in text
\titleformat{\section}[hang]
	{\normalfont\scshape\color{eurgreen}\raggedright}
	{\normalfont\scshape\color{eurgreen}\thesection}
	{1em}
	{\textls[90]{#1}}

% subsection formatting in text
\titleformat{\subsection}[hang]
	{\normalfont\color{eurgreen}\itshape\raggedright}
	{\normalfont\thesubsection}
	{.5em}
	{#1}
	[\vspace{0.5em}]

% section formatting in toc
\titlecontents{section}
	[3.8em]
	{\addvspace{0.75em}}
	{\contentslabel{2.3em}\itshape}
	{\hspace*{-2.3em}}
	{\hspace*{5pt}\titlerule*[1pc]{.}\hspace*{2pt}\contentspage}[\addvspace{0.1em}]

% list of figures formatting
\titlecontents{figure}
	[2.8em]
	{\addvspace{1.0em}}
	{\contentslabel{2.8em}}
	{\hspace*{0em}}
	{\hspace*{5pt}\titlerule*[1.0pc]{.}\hspace*{2pt}\contentspage}[]

% list of tables formatting
\titlecontents{table}
	[2.8em]
	{\addvspace{1.0em}}
	{\contentslabel{2.8em}}
	{\hspace*{0em}}
	{\hspace*{5pt}\titlerule*[1.0pc]{.}\hspace*{2pt}\contentspage}[]

%list of algorithms formatting
\makeatletter
\renewcommand*{\listof}[2]{%
	\@ifundefined{ext@#1}{\float@error{#1}}{%
		\@namedef{l@#1}{\l@figure}\float@listhead{#2}%
		\begingroup\setlength{\parskip}{\z@}%
			\@starttoc{\@nameuse{ext@#1}}%
		\endgroup}}
\makeatother
