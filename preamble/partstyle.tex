% # vim: ft=tex

% Initially clear the partimage
\def\partimage{}

\titleformat{\part}[display]
	{\vfill}
	{\color{eurgreen}\centering{\fontsize{80}{96}\normalfont\thepart}}
	{20pt}
	{\centering\color{eurgreen}\scshape\Huge\textls[80]{#1}%
	{%
		\vfill
		\ifx\partimage\empty
			{}
		\else
			\includegraphics[height=.18\textheight]{\partimage}%
		\fi
		\vfill
	}
}

\titlecontents{part}%
	[0.2em]
	{\addvspace{2.0em}\color{eurgreen}\scshape}
	{Part~\thecontentslabel:~}%\contentslabel{1.3em}}
	{\hspace*{-1.3em}}
	{\titlerule*[.5pc]{}\contentspage}
	[\addvspace{.5em}]
