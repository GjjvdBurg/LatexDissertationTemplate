\fancypagestyle{frontback}{%
	\fancyhf{}
	\renewcommand*{\chaptermark}[1]{\markboth{##1}{}}
	\renewcommand{\headrulewidth}{0pt}%
	\fancyhead[RO]{%
		{\small\itshape\nouppercase{\leftmark}}%
	}
	\fancyfoot[LE]{\small\thepage}
	\fancyfoot[RO]{\small\thepage}
}

\fancypagestyle{mainmatter}{%
	\fancyhf{}
	\renewcommand*{\chaptermark}[1]{\markboth{##1}{}}
	\renewcommand{\headrulewidth}{0pt}%
	\fancyhead[RO]{%
		{\small\itshape\thechapter.\hspace{.5em}\nouppercase{\leftmark}}%
	}
	\fancyfoot[LE]{\small\thepage}
	\fancyfoot[RO]{\small\thepage}
}

\fancypagestyle{bibliography}{%
	\fancyhf{}
	\renewcommand*{\chaptermark}[1]{\markboth{##1}{}}
	\renewcommand*{\sectionmark}[1]{\markright{##1}}
	\renewcommand{\headrulewidth}{0pt}%
	\fancyhead[RO]{%
		{\small\itshape\nouppercase{\leftmark}}%
	}
	\fancyfoot[LE]{\small\thepage}
	\fancyfoot[RO]{\small\thepage}
}

\fancypagestyle{plain}{%
	\fancyhf{}
	\renewcommand{\headrulewidth}{0pt}%
	\renewcommand{\footrulewidth}{0pt}
	\fancyfoot[LE]{
		\small\thepage
	}
	\fancyfoot[RO]{
		\small\thepage
	}
}
