\fancypagestyle{mainmatter}{%
	\fancyhf{}
	\renewcommand*{\chaptermark}[1]{\markboth{##1}{}}
	\renewcommand*{\sectionmark}[1]{\markright{##1}}
	\renewcommand{\headrulewidth}{0pt}%

	\fancyhead[LE]{\small\scshape\thepage\hspace{2em}\nouppercase{\leftmark}}
	\fancyhead[RO]{\small\scshape\nouppercase{\rightmark}\hspace{2em}\thepage}
}

	\fancyhf{}
	\renewcommand*{\chaptermark}[1]{\markboth{##1}{}}

	\fancyhead[LE]{\small\scshape\thepage\hspace{2em}\nouppercase{\leftmark}}
